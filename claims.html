// Register new claim (dummy – adds to in‑memory data) const registerClaimForm = document.getElementById("registerClaimForm"); registerClaimForm.addEventListener("submit", (e) => { e.preventDefault(); const policyNo = document.getElementById("policyNoInput").value.trim(); const lossDate = document.getElementById("lossDateInput").value; const lossDesc = document.getElementById("lossDescInput").value.trim(); const policy = window.data.policies.find(p => p.policyNo === policyNo); if (!policy) { alert("Policy not found. Please check the policy number."); return; } const newClaimNo = `CLM-${Date.now().toString(36).slice(2, 6).toUpperCase()}`; const newClaim = { id: window.data.claims.reduce((max, c) => Math.max(max, c.id), 0) + 1000, claimNo: newClaimNo, policy: policy.id, lossDate: lossDate, status: "Registered", notes: lossDesc }; window.data.claims.push(newClaim); newClaimForm.reset(); newClaimForm.style.display = "none"; renderClaimsTable(); }); // Claims table body const claimsTableBody = document.querySelector("#claimsTable tbody"); function renderClaimsTable() { // Clear tbody claimsTableBody.innerHTML = ""; window.data.claims.forEach(claim => { const policy = window.data.policies.find(p => p.id === claim.policy) || {}; const tr = document.createElement("tr"); tr.innerHTML = [ `${claim.claimNo}`, `${policy.policyNo || "Unknown"}`, `${claim.lossDate}`, `${claim.status}`, ` ${(app.currentSession?.role === "ClaimsHandler" || app.currentSession?.role === "BranchManager") ? `` : ""} ` ].join(""); claimsTableBody.appendChild(tr); }); // Dummy click handlers for View/Approve claimsTableBody.querySelectorAll("button").forEach(btn => { btn.addEventListener("click", (e) => { const claimId = e.target.getAttribute("data-claim"); const claim = window.data.claims.find(c => c.id === parseInt(claimId)); alert(`Viewing/Approving claim: ${claim?.claimNo} (status: ${claim?.status})`); }); }); } function getStatusClass(status) { switch (status) { case "Registered": return "status-yellow"; case "Assessing": return "status-orange"; case "Approved": return "status-green"; case "Closed": return "status-blue"; case "Repudiated": return "status-red"; default: return "status-gray"; } } renderClaimsTable(); });
